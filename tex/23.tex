		\begin{thm}[Constant Multiple Rule]
			If $c$ is a constant and $f$ is a differentiable function, then
			\showto{ins}{
				\[\dfrac{d}{dx}[c\cdot f(x)] = c\cdot\dfrac{d}{dx}[f(x)]\]
			}
			\showto{st}{
				\vspace{.25in}
			}
		\end{thm}
			
		There are three more important rules:
		\begin{thm}[Sum/Difference Rule]
			If $f$ and $g$ are differentiable functions, then 
			\showto{ins}{
				\[\dfrac{d}{dx}[f(x)\pm g(x)] = \dfrac{d}{dx}[f(x)]\pm \dfrac{d}{dx}[g(x)]\]
			}
			\showto{st}{
				\vspace{.5in}
			}
		\end{thm}
			\vs{.25}
			
		\begin{question}
			Restate the sum/difference rule in words.
		\end{question}
			\vs{1}
			\newpage
			
		\begin{thm}[Product Rule]
			If $f$ and $g$ are both differentiable, then
			\showto{ins}{
				\[\dfrac{d}{dx}[f(x)\cdot g(x)] = \dfrac{d}{dx}[f(x)]\cdot g(x) + f(x)\cdot \dfrac{d}{dx}[g(x)]\]
			}
			\showto{st}{
				\vspace{.5in}
			}
		\end{thm}
			\vs{.25}
			
		\begin{ex}
			Compute the derivative of $f(x) = x^3\cdot x^5$
		\end{ex}
			\vs{1} 

		\begin{thm}[Quotient Rule]
			If $f$ and $g$ are both differentiable (with $g\neq 0$), then
			\showto{ins}{
				\[\dfrac{d}{dx}\left[\dfrac{f(x)}{g(x)}\right] = \dfrac{\dfrac{d}{dx}[f(x)]\cdot g(x) - f(x)\cdot \dfrac{d}{dx}[g(x)]}{(g(x))^2}\]
			}
			\showto{st}{
				\vspace{.75in}
			}
		\end{thm}
			\vs{.25}

		\begin{ex}
			Compute the derivative of $g(x) = \dfrac{x}{x+1}$
		\end{ex}
			\vs{1}
