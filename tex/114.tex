	\subsection*{Before Class}
	\subsubsection*{The Direct Comparison Test}
		\begin{ex} Let $a_n = \dfrac{1}{n^2}$, $b_n = \dfrac{1}{n^2+1}$.
			\begin{enumerate}[(a)]
				\item Find the first ten terms of $a_n$, and the first ten terms of $b_n$.
					\vs{1}
				\item Represent the first four terms of $a_n$ and $b_n$ as rectangles, and plot them on the same chart.
					\vs{2}
				\item What can you say about the total area of the rectangles from $a_n$ in comparison to the total area of the rectangles from $b_n$?
					\vs{.5}
			\end{enumerate}
		\end{ex}
			\newpage
			
		\begin{rmk}[The Direct Comparison Test]
			Suppose that $\ds \sum a_n$ and $\ds \sum b_n$ are series with positive terms.  Then,\\
			\begin{itemize}
				\setlength\itemsep{20pt}
				
				\item If $\ds \sum b_n$ converges and \blank{1.5}, then \blank{2.3}.
				\item If $\ds \sum b_n$ diverges and \blank{1.5}, then \blank{2.4}.
			\end{itemize}
		\end{rmk}

		\begin{rmk}[Algebra in Comparing Series]
			\begin{itemize}
				\setlength\itemsep{15pt}
				\item Removing a $+$ from the numerator makes the fraction \blank{2}.
				\item Removing a $+$ from the denominator makes the fraction \blank{2}.
				\item Removing a $-$ from the numerator makes the fraction \blank{2}.
				\item Removing a $-$ from the denominator makes the fraction \blank{2}.
			\end{itemize}\\
			Be careful$-$you only want to remove \emph{one type} of sign at a time!
		\end{rmk}

		\begin{rmk}[The Limit Comparison Test]
			Let $\ds \sum a_n$ and $\ds \sum b_n$ be series with positive terms.  Let $\ds \lim_{n\to\infty} \dfrac{a_n}{b_n} = c$, where $0 < c < \infty$.  Then\\[20pt] \blank{5}.
		\end{rmk}
			\newpage
		
		\begin{ex}
			Determine if the series $\ds \sum \dfrac{3n^2 + n}{\sqrt{3 + n^5}}$ converges or diverges.
		\end{ex}
