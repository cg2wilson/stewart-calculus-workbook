\begin{align*} \ds \int_0^\infty \dfrac{1}{1+x^2}\, dx \amp = \lim_{t\to\infty} \int_0^t \dfrac{1}{1+x^2}\, dx\\ \ds \lim_{t\to\infty} \int_0^t \dfrac{1}{1+x^2}\, dx \amp = \lim_{t\to \infty} \lrpar{\inv{\tan}(x)}\bigg\rvert_0^t\\ \ds \lim_{t\to \infty} \lrpar{\inv{\tan}(x)}\bigg\rvert_0^t \amp = \lim_{t\to \infty} \lrpar{\inv{\tan}(t)-\inv{\tan}(0)}\\ \amp = \dfrac{\pi}{2} \end{align*}
